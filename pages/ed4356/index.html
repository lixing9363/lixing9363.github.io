    # 416. 分割等和子集 (opens new window)

    解法一:

    查看答案 
    class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        sums = sum(nums)
        if sums % 2 != 0:
            return False
        harf = sums / 2
        memo = dict()
        def dp(nums,i,harf):
            # 定义函数dp(nums,i,harf): 表示nums[:i] 能不能凑出harf
            if i == 1:
                if nums[0] == harf:
                    return True
                return False
            if (i,harf) in memo:
                return memo[(i,harf)]
            # 第一种情况,如果harf<nums[i-1] 则肯定不能用nums[i-1]凑
            if harf<nums[i-1]:
                res = dp(nums,i-1,harf)
            else:
            # 第二种情况: 用nums[i-1]凑或者不用,取or结果
                res = dp(nums,i-1,harf-nums[i-1]) or dp(nums,i-1,harf)
            memo[(i,harf)] = res
            return res
        
        return dp(nums,len(nums),harf)

    查看答案 
    class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        sums = sum(nums)
        if sums % 2 != 0:
            return False
        
        n = len(nums)
        harf = sums / 2
        # 定义dp[i][j],表示nums[0:i]能不能凑出和为j
        dp = [ [False] * int(harf + 1) for _ in range(n+1) ]

        # 初始化可以确定的值
        # 当j为0时,肯定都能凑出
        for i in range(n+1):
            dp[i][0] = True
        
        # 子顶向下开始迭代
        for i in range(1,n+1):
            for j in range(1,int(harf + 1)):
                
                # 第一种情况,如果j < nums[i-1] 则肯定不能用nums[i-1]凑
                if j < nums[i-1] :
                    dp[i][j] = dp[i-1][j]
                else:
                # 第二种情况,用nums[i-1]凑或者不用,取or结果
                    dp[i][j] = dp[i-1][j] or dp[i-1][j-nums[i-1]]

        
        return dp[n][int(harf)]

    # 518. 零钱兑换 II (opens new window)

    查看答案 
    class Solution:
    def change(self, amount: int, coins: List[int]) -> int:
        memo = {}

        def dp(coins,i,amount):
            # 定义函数dp(coins,i,amount)表示coins[:i]凑出amount的组合数
            if i == 1:
                if amount % coins[0] == 0:
                    return 1
                else:
                    return 0
            if (i,amount) in memo:
                return memo[(i,amount)]
            # 如果amount < coins[i-1], 则不能用coins[i-1]凑
            if amount < coins[i-1]:
                res = dp(coins,i-1,amount)
            # 否则可以用coins[i-1]凑,也可以不用
            else:
                # 不用coins[i-1]凑
                res = dp(coins,i-1,amount)
                # 用coins[i-1]凑也有很多方式,可以用1个,可以用2个....n个
                target = amount
                n = 1
                while target >= coins[i-1]:
                    target = amount - n * coins[i-1]
                    res += dp(coins,i-1,target)
                    n += 1
            memo[(i,amount)] = res
            return res
        rs = dp(coins,len(coins),amount)
        return rs
