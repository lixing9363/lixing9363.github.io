    class Upload(View):
    def post(self, request):
        file = request.FILES['myfile']
        username = request.POST.get('username')
        filepath = os.path.join(settings.MEDIA_ROOT, file.name)
        with open(filepath, 'wb') as f:
            data = file.file.read()
            f.write(data)
        print(username)
        return HttpResponse('文件上传成功')
