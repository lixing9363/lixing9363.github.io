    # 19. 删除链表的倒数第 N 个结点 (opens new window)

    查看答案 
    # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def removeNthFromEnd(self, head: ListNode, n: int) -> ListNode:
        fast = slow = head
        while n>= 0:
            if not fast:
                return head.next
            fast = fast.next
            n -= 1
        while fast:
            fast = fast.next
            slow = slow.next
        slow.next = slow.next.next
        return head

    # 26. 删除有序数组中的重复项 (opens new window)

    查看答案 
    class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        if not nums:
            return 0
        fast = slow = 0
        while fast <= len(nums) -1 :
            if nums[fast] == nums[slow]:
                fast += 1
            else:
                slow += 1
                nums[slow] = nums[fast]
                fast += 1
        
        return slow + 1

    # 283. 移动零 (opens new window)

    查看答案 
    class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        fast = slow = 0
        while fast < len(nums):
            if nums[fast] == 0:
                fast += 1
            else:
                nums[slow] = nums[fast]
                fast += 1
                slow += 1
        
        for i in range(slow,len(nums)):
            nums[i] = 0

# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def deleteDuplicates(self, head: ListNode) -> ListNode:
        if not head:
            return head
        fast = slow = head
        while fast:
            if fast.val == slow.val:
                fast = fast.next
            else:
                slow.next = fast
                slow = slow.next
                fast = fast.next
        slow.next = fast
        return head

#     def __init__(self, x):
#         self.val = x

class Solution:
    def hasCycle(self, head: ListNode) -> bool:
        fast = slow = head
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if fast == slow:
                return True
        return False

    # 142. 环形链表 II (opens new window)

    快慢指针追及

    # 分析

    此题还是看作一个追及问题,把它转化为一个新的问题:

    假设有fast和slow两个人,slow每次走一步,fast每次走两步. 两人从head起点同时起步走, 假如跑道是直线, 则当一个人走到终点时, 返回None. 如果跑道走了一段直线后,变成了环形,则返回开始入环的节点.

    再分析这个新问题:

    1. 当跑道是直跑道时,很好判断

    2. 当跑道带环时,则复杂些,我们再将这种情况转化为数学问题:

      • 假设,当两人走了k步之后相遇了,那么slow走了k步,fast走了2k步.

        两人相差k步,则说明环形段一圈的长度为k/n步,也就是说一圈长度的n倍就是k

        • 这里的情况需要说明一下:当两人相遇时,不能确定fast已经在圈里面走了多少圈了,因为有可能fast很早就进圈了,而slow还没有进圈,所有fast可能走了n圈了.也就是说,k是圈长的整数倍,如果从环的任何一个地方开始走,只要走k步,那么必然回到原点

    重点来了

    再假设,在相遇的这个时刻,开始调整,让fast的速度变为每次一步,slow的速度不变任然每次一步.

    fast的位置不变,还是在相遇点,slow的位置重新回到起点head.同时开始走.

    那么当slow走k步后,很显然,slow仍然是在相遇点(速度路程都没变)

    而fast在走了k步后,很显然fast也是还在相遇点(fast在环内走,而且k是环长的倍数)

    所以可以得到,这个时候他们相遇了.

    那么重点来了,他们这是第一次相遇吗?

    不是! 他们两的速度是一样的,都是每次一步,slow先走直线,再入环. fast一直在环里面.

    假如我们让他们都以每次一步的速度后退,会发现,他们是在入环口开始第一次相遇的.

    那么结论就出来了:

    让slow和fast两人都以每次一步的速度,分别从起点head和相遇点前进,当他俩相遇的地方,就是入环点

    查看答案 
    # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: ListNode) -> ListNode:
        fast = slow = head
        cir = False
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if fast == slow:
                cir = True
                break
        
        if not cir:
            return None
        
        fast = head
        while fast != slow:
            fast = fast.next
            slow = slow.next
        return slow

    # 876. 链表的中间结点 (opens new window)

    查看答案 
    # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def middleNode(self, head: ListNode) -> ListNode:
        fast = slow = head
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
        return slow
