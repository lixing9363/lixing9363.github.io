# 一对一映射

  • 定义model:一个人对应一个身份证
class UserProfile(models.Model):
    name = models.CharField(max_length=16, verbose_name='用户名')

    class Meta:
        db_table = 'UserProfile'
        verbose_name = '用户'
        verbose_name_plural = verbose_name

    def __str__(self):
        return '%s' % (self.name)


class IDCard(models.Model):
    id_card = models.CharField(max_length=18, verbose_name='身份证号')
    user = models.OneToOneField(UserProfile, unique=True, on_delete=models.CASCADE)

    class Meta:
        db_table = 'IDCard'
        verbose_name = '身份证号'
        verbose_name_plural = verbose_name

    def __str__(self):
        return '%s' % (self.id_card)

# 一对多映射

如一个书有一个作者,一个作者有多本书..

class Author(models.Model):
    name = models.CharField(max_length=16, verbose_name='作者名')

    class Meta:
        db_table = 'Author'
        verbose_name = '作者'
        verbose_name_plural = verbose_name

    def __str__(self):
        return '%s' % (self.name)


class Book(models.Model):
    name = models.CharField(max_length=16, verbose_name='书名')
    author = models.ForeignKey(Author, on_delete=models.CASCADE)

    class Meta:
        db_table = 'Book'
        verbose_name = '书名'
        verbose_name_plural = verbose_name

    def __str__(self):
        return '%s' % (self.name)

  • model.py

    class Article(models.Model):
    name = models.CharField(max_length=16, verbose_name='文章')

    class Meta:
        db_table = 'Article'
        verbose_name = '文章'
        verbose_name_plural = verbose_name

    def __str__(self):
        return '%s' % (self.name)


class Publication(models.Model):
    name = models.CharField(max_length=16, verbose_name='出版社')
    book = models.ManyToManyField(Article)

    class Meta:
        db_table = 'Publication'
        verbose_name = '出版社'
        verbose_name_plural = verbose_name

    def __str__(self):
        return '%s' % (self.name)

一对多或者多对多情况下,要在admin后台显示'多'的属性:

admin.py

class ArticleManager(admin.ModelAdmin):
    def 出版社(self, obj):
        return [a.name for a in obj.publication_set.all()]
    list_display = ['name','出版社']

class PublicationManager(admin.ModelAdmin):
    def 文章(self, obj):
        return [a.name for a in obj.article.all()]
    list_display = ['name','文章']
