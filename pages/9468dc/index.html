    class Solution:
    def coinChange(self, coins, amount: int) -> int:
        def ncc(m):
            if m == 0:
                return 0
            if m < 0:
                return -1
            # 求最小值,初始化一个不可能的最小值
            res = amount + 1
            for coin in coins:
                ncc_res = ncc(m - coin)
                if ncc_res == -1:
                    pass
                else:
                    res = min(res, 1 + ncc_res)
            return res if res != amount + 1 else -1

        return ncc(amount)

# 三. 带备忘录的递归

在暴力递归的情况下,如果不加备忘录,会存在很多冗余的递归,我们可以将一些已经算出来的结果存在备忘录中,这样可以减少冗余的递归计算,避免超时

    def coinChange(self, coins: List[int], amount: int) -> int:
        memo = dict() # 定义一个备忘录
        def ncc(m):
            # 如果在备忘录中,则不需要再计算了
            if m in memo:
                return memo[m]
            if m == 0:
                return 0
            if m < 0:
                return -1
            # 求最小值,初始化一个不可能的最小值
            res = amount + 1
            for coin in coins:
                ncc_res = ncc(m - coin)
                # 判断去掉的那个硬币是否合理
                if ncc_res == -1:
                    # 如果不合理,则res就还是那个不合理的值
                    pass
                else:
                    # 如果合理,就是合理的所需硬币数
                    res = min(res, 1 + ncc_res)
            # 如果循环计算后的值还是不合理,就是-1
            res = res if res != amount + 1 else -1
            # 存入备忘录
            memo[m] = res
            return res

        return ncc(amount)
