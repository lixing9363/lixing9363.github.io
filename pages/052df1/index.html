# 2. 两数相加 (opens new window)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        rs = head  = ListNode(0)
        up = 0
        while l1 or l2:
            v1 = l1.val if l1 else 0
            v2 = l2.val if l2 else 0
            sum1 = v1 + v2+up
            if sum1 >= 10:
                up = 1
                rs.next = ListNode(sum1-10)
            else:
                up = 0
                rs.next = ListNode(sum1)
            rs = rs.next
            l1 = l1.next if l1 else l1
            l2 = l2.next if l2 else l2
        if up == 1:
            rs.next = ListNode(1)
        return head.next

# 344. 反转字符串 (opens new window)

class Solution:
    def reverseString(self, s: List[str]) -> None:
        """
        Do not return anything, modify s in-place instead.
        """
        left = 0 
        right = len(s) - 1
        while left < right:
            s[left],s[right] = s[right],s[left]
            left += 1
            right -= 1

# 5. 最长回文子串 (opens new window)

笔记

  1. 找回文，从中间开始
  2. 判断回文，从两边开始

但是这样会出现两种情况：

  1. 回文长度是奇数时,是从中间点开始向两边张开
  2. 回文长度是偶数时,是从中间点i和i+1向两边扩张

将两种情况都找出,作对比,选择长的

class Solution:
    def longestPalindrome(self, s: str) -> str:
        # 找回文是从中间向两边扩散
        # 中间点的指针为i
        rs = s[0]
        for i in range(len(s)-1):
            str1 = self.find_palindrom(s,i,i)
            str2 = self.find_palindrom(s,i,i+1)
            rs = str1 if len(str1) > len(rs) else rs
            rs = str2 if len(str2) > len(rs) else rs
        return rs

    

    def find_palindrom(self, s, left, right):
        # 自定义一个方法
        # 用两个指针left和right从s字符串中找回文
        rs = s[0]
        while left >= 0 and right <= len(s) -1:
            if s[left] != s[right]:
                break
            rs = s[left:right+1]
            left -= 1
            right += 1
        # print(rs)
        return rs

# 4. 寻找两个正序数组的中位数 (opens new window)

class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        nums = nums1 + nums2
        nums.sort()
        left = 0
        right = len(nums) -1 
        rs = 0
        while left <= right:
            mid_val = (nums[left] + nums[right])/2
            rs = mid_val
            left += 1
            right -= 1
        return rs
