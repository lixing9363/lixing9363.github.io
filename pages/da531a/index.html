    # 704. 二分查找 (opens new window)

    查看答案 
    class Solution:
    def search(self, nums: List[int], target: int) -> int:
        left = 0
        right = len(nums) - 1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:
                return mid
            elif nums[mid] > target:
                right = mid - 1
            else:
                left = mid + 1
        return -1

    # 34. 在排序数组中查找元素的第一个和最后一个位置 (opens new window)

    查看答案 
    class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        if not nums:
            return [-1,-1]
        left = self.left_bound(nums,target)
        right = self.right_bound(nums, target)
        return [left,right]
    
    def left_bound(self,nums,target):
        left = 0
        right = len(nums) - 1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] >= target:
                right = mid - 1
            else:
                left = mid + 1
        if left >= len(nums) or nums[left] != target:
            return -1
        return left
    
    def right_bound(self,nums,target):
        left = 0
        right = len(nums)-1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] <= target:
                left = mid + 1
            else:
                right = mid - 1
        if right < 0 or nums[right] != target:
            return -1
        return right

    # 875. 爱吃香蕉的珂珂 (opens new window)

    查看答案 
    class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:

        def eat_end(v):
            # 定义函数, 在速度为v时能否吃完香蕉
            t = 0
            for pile in piles:
                if pile % v == 0:
                    t += pile // v
                else:
                    t += pile //v +1
            return t <= h
        
        min_v = 1
        max_v = max(piles)
        while min_v < max_v:
            mid_v = (min_v+max_v) // 2
            if eat_end(mid_v):
                max_v = mid_v
            else:
                min_v = mid_v + 1
        return min_v

    # 1011. 在 D 天内送达包裹的能力 (opens new window)

    查看答案 
    class Solution:
    def shipWithinDays(self, weights: List[int], days: int) -> int:
        min_cap = max(weights)
        max_cap = sum(weights)
        while min_cap < max_cap:
            mid_cap = (min_cap + max_cap) // 2
            if self.can_finish(weights,days,mid_cap):
                max_cap = mid_cap 
            else:
                min_cap = mid_cap + 1
        return min_cap



    def can_finish(self,weights,days,per_day):
        # 定义函数, 每天最低运载能力为per_day, 能不能完成
        i = 0
        d = 1
        remain_cap = per_day # 当天剩余运载能力
        while i < len(weights):
            if remain_cap >= weights[i]:
                remain_cap -= weights[i]
                i += 1
            else:
                remain_cap = per_day
                d += 1
        # print(d)
        return d <= days

    # 410. 分割数组的最大值 (opens new window)

    查看答案 
    class Solution:
    def splitArray(self, nums: List[int], m: int) -> int:
        min_sum = max(nums)
        max_sum = sum(nums)
        while min_sum < max_sum:
            mid_sum = (min_sum + max_sum) // 2
            if self.split(nums,mid_sum) > m:
                min_sum = mid_sum + 1
            else:
                max_sum = mid_sum
        return min_sum
    

    def split(self, nums, max_sum):
        # 定义函数, 表示将数组分割的子数组和最大为max_sum, 可以分成几个
        sum1 = 0
        count = 1
        for num in nums:
            if sum1 + num <= max_sum:
                sum1 += num
            else:
                sum1 = num
                count += 1
        # print(count)
        return count
