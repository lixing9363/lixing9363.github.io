1.需要父传给子组件的参数,需要使用props参数,添加改参数;

且需要在使用处引用该参数(:data="tableData",:height="ht")

2.父引用子组件时,需要将传递的参数在标签中赋值:

  • 如果是自定义的属性,则直接用:加上props中的名称赋值即可(:msg="message",:ht="height")

  • 如果是特定的属性,则不是使用自定义的赋值方法(用:table-data="tData",而不是:tableData="tData")

    原因是使用驼峰式命名会自动转化为table-data

# 3.父传给子,子改变后再传给父

  • my_select.html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
    <!-- 引入样式 -->
    <link rel="stylesheet" href="../element-ui/lib/theme-chalk/index.css">
    <!-- 引入组件库 -->
    <script src="../element-ui/lib/index.js"></script>
    <script src="../components/my_select.js"></script>
    <title>my_select</title>
</head>
<body>

<div id="app">
    <my_select :options="options" @on_submit_father_prop="onSubmitFather" @on_change_father_prop="onSubmitFather"></my_select>
    <hr>
    父组件value:{{value}}
</div>

<script>
    new Vue({
        el: '#app',
        data(){
            return {
                options: [ '黄金糕', '双皮奶','蚵仔煎','龙须面'],
                value:""
            }
        },
        methods:{
            onSubmitFather(a){
                this.value = a
            }
        }
    })
</script>

</body>
</html>

  • my_select.js
Vue.component('my_select', {
    template:
        `<div>
            <el-form :inline="true"  class="demo-form-inline">
              <el-form-item label="活动区域">
                <el-select v-model="value" placeholder="请选择" @change="onChange">
                    <el-option
                      v-for="item in options"
                      :key="item"
                      :label="item"
                      :value="item">
                    </el-option>
                  </el-select>
              </el-form-item>
              <el-form-item>
                <el-button type="primary" @click="onSubmit">查询</el-button>
              </el-form-item>
            </el-form>
            <hr>
            子组件value:{{value}}
        </div>`,
    data(){
        return{
            value:''
        }
    },
    props:['options'],
    methods: {
        onSubmit(){
            this.$emit('on_submit_father_prop', this.value)
        },
        onChange(){
            this.$emit('on_change_father_prop', this.value)
        }
    },
})
