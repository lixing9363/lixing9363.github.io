    # 226. 翻转二叉树 (opens new window)

    查看答案

    此题前序,后序遍历都可以

    # Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def invertTree(self, root: TreeNode) -> TreeNode:
        if not root:
            return
        self.invertTree(root.left)
        self.invertTree(root.right)
        root.left , root.right = root.right , root.left
        return root

    # 114. 二叉树展开为链表 (opens new window)

    查看答案 
    # Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def flatten(self, root: TreeNode) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        if  not root:
            return
        self.flatten(root.left)
        self.flatten(root.right)

        left_node = root.left
        right_node = root.right

        root.left = None
        root.right = left_node
        left_end_node = root
        while left_end_node.right:
            left_end_node = left_end_node.right
        left_end_node.right = right_node

    # 116. 填充每个节点的下一个右侧节点指针 (opens new window)

    将两个跨父级节点的同一层节点连接起来,那么该怎么办呢?

    可以看到,如果只安排一个节点,很难将两个跨父级节点连起来,那么一个节点不行,就用两个

    查看答案 
    """
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if root:
            self.connect_two(root.left,root.right)
        return root
        
    
    def connect_two(self,left_node,right_node):
        if left_node and right_node:
            left_node.next = right_node
            self.connect_two(left_node.left,left_node.right)
            self.connect_two(right_node.left,right_node.right)
            self.connect_two(left_node.right,right_node.left)

    # 652. 寻找重复的子树 (opens new window)

    怎么对比两个节点的树相同呢?这是关键

    假如我们把一个节点的左边和右边转成字符串(或者列表等,任何你觉得方便比较的类型), 然后和另一个节点这样转化的进行比较,就可以知道是否一样了

    查看答案 
    # Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def __init__(self):
        self.memo = {}
        self.res = []

    def findDuplicateSubtrees(self, root: Optional[TreeNode]) -> List[Optional[TreeNode]]:
        if not root:
            return
        root_str = self.to_str(root)
        if root_str in self.memo:
            if self.memo[root_str] == 0:
                self.res.append(root)
            self.memo[root_str] += 1
        else:
            self.memo[root_str] = 0
        self.findDuplicateSubtrees(root.left)
        self.findDuplicateSubtrees(root.right)
        return self.res

    
    def to_str(self,node):
        if node == None:
            return "#"
        left_str = self.to_str(node.left)
        right_str = self.to_str(node.right)
        return str(node.val) + ";" + str(left_str) + ";" + str(right_str)

    # 654. 最大二叉树 (opens new window)

    • 题目的意思
    1. 就是比如输入一个数组nums = [3,2,1,6,0,5],找到最大值为6,那么6就是根节点
    2. 根节点左边就是[3,2,1]数组,右边就是[0,5]
    3. 然后左边的最大值是3,那么3就是左边的根节点,3节点的左边没有,就是None;右边就是[2,1]组成的节点,也是这样的规则
    查看答案 
    # Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def constructMaximumBinaryTree(self, nums: List[int]) -> TreeNode:
        max_index = self.get_max_index(nums)
        root = TreeNode(nums[max_index])
        if max_index != 0:
            root.left = self.constructMaximumBinaryTree(nums[:max_index])
        if max_index != len(nums) - 1:
            root.right = self.constructMaximumBinaryTree(nums[max_index+1:])
        return root

    

    def get_max_index(self, nums):
        # 找到最大值所在的索引
        max_index = 0
        for i in range(len(nums)):
            if nums[i] > nums[max_index]:
                max_index = i
        return max_index
