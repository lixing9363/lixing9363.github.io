# html

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
    <!-- 引入样式 -->
    <link rel="stylesheet" href="../static/plugin/element-ui/lib/theme-chalk/index.css">
    <!-- 引入组件库 -->
    <script src="../static/plugin/element-ui/lib/index.js"></script>
    <script src="../static/js/base_js.js"></script>

</head>
<body>
<div id="app">
    <!--header-->
    <el-header>
        <my_header :active-index="headerActiveIndex"></my_header>
    </el-header>
    <el-container class="container">
        <!--aside-->
        <my_aside :active-index="asideActiveIndex"></my_aside>
        <el-container>
            <el-main>Main</el-main>
        </el-container>
    </el-container>
    <el-footer><my_foot></my_foot></el-footer>
</div>
<script>
    var vm = new Vue({
        el: "#app",
        data() {
            return {
                headerActiveIndex: '1',
                asideActiveIndex: '1',
            }
        }
    })
</script>
</body>
</html>

Vue.component('my_header', {
    template:
        `<div>
            <el-menu
                :default-active="activeIndex"
                class="el-menu-demo"
                mode="horizontal"
                background-color="#545c64"
                text-color="#fff"
                active-text-color="#ffd04b">
            <div style="width: 100px;height: 30px;background: red;float: left;margin-top: 15px;text-align: center;margin-right: 100px">logo</div>
            <el-menu-item index="1" @click="goTo('/testHtml/page/daohang1/daohang1.html')">导航1</el-menu-item>
            <el-menu-item index="2">导航2</el-menu-item>
            <el-menu-item index="3">导航3</el-menu-item>
            <el-menu-item style="float: right">信息</el-menu-item>
        </el-menu>
</div>`,
    props: ['activeIndex'],
    methods:{
        goTo(val){
            location.href = val
        }
    }
})

Vue.component('my_aside', {
    template:
        `<div>
            <el-aside width="auto" style="max-width: 300px;min-width: 200px">
            <el-menu
                    :default-active="activeIndex"
                    class="el-menu-vertical-demo">
                <el-menu-item index="1">
                    <i class="el-icon-menu"></i>
                    <span slot="title">导航一</span>
                </el-menu-item>
                <el-menu-item index="2">
                    <i class="el-icon-menu"></i>
                    <span slot="title">导航二</span>
                </el-menu-item>
                <el-menu-item index="3" disabled>
                    <i class="el-icon-document"></i>
                    <span slot="title">导航三</span>
                </el-menu-item>
                <el-menu-item index="4">
                    <i class="el-icon-setting"></i>
                    <span slot="title">导航四</span>
                </el-menu-item>
            </el-menu>
        </el-aside>
</div>`,
    props: ['activeIndex'],
    methods:{
        goTo(val){
            location.href = val
        }
    }
})
