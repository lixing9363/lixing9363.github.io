  • ChildComponent.vue

    <template>
  <div>
    <button type="button" @click="sendToFather">子组件数据传给父组件</button>
  </div>
</template>

<script>
export default {
  name: 'ChildComponent',
  data(){
    return {
      childData1:'子传父的数据'
    }
  },
  methods:{
    sendToFather(){
      this.$emit('取个名字',this.childData1)
    }
  }
}
</script>

  • FatherComponent.vue

    <template>
  <div>
    {{ fatherData }}
    <ChildComponent @取个名字='childToFather'></ChildComponent>
  </div>
</template>

<script>
import ChildComponent from "@/components/ChildComponent.vue";

export default {
  name: 'FatherComponent',
  components: {ChildComponent: ChildComponent},
  data(){
    return{ fatherData: "" }
  },
  methods:{
    childToFather(msg){
      this.fatherData = msg
    }
  }
}
</script>
