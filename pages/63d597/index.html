    # 300. 最长递增子序列 (opens new window)

    查看答案 
    class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        memo = {}
        def dp(i):
            # 定义函数dp(i), 表示在nums[:i+1]中,以nums[i]为结尾的子序列长度
            if i == 0:
                return 1
            if i < 0:
                return 0
            if i in memo:
                return memo[i]
            rs = 1
            rs_list = []
            for j in range(i,-1,-1):
                if nums[j] < nums[i]:
                    rs_list.append(dp(j))
            if rs_list:
                rs = rs + max(rs_list)
            memo[i] = rs
            return rs
        
        rs_list = []
        for i in range(len(nums)):
            rs_list.append(dp(i))
        # print(rs_list)
        return max(rs_list)

    解法一: 会超时

    查看答案 
    class Solution:
    def maxEnvelopes(self, envelopes: List[List[int]]) -> int:
        envelopes.sort()
        memo = {}
        def dp(i):
            # 定义函数dp(i):表示以envelopes[i]为最大的信封的情况下, 最多的套娃数
            if i == 0:
                return 1
            if i in memo:
                return memo[i]
            rs = 1
            for j in range(i,-1,-1):
                if envelopes[j][0] < envelopes[i][0] and envelopes[j][1] < envelopes[i][1]:
                    rs = max(rs, dp(j) + 1)
            memo[i] = rs
            return rs
        
        rs_list = []
        for i in range(len(envelopes)):
            rs_list.append(dp(i))
        # print(envelopes)
        # print(memo)
        return max(rs_list)

    查看答案 
    class Solution:
    def maxEnvelopes(self, envelopes: List[List[int]]) -> int:
        # 这里将信封排序, 
        envelopes.sort(key=lambda x: (x[0], -x[1]))
        memo = {}
        def dp(i):
            # 定义函数dp(i):表示以envelopes[i]为最大的信封的情况下, 最多的套娃数
            if i in memo:
                return memo[i]
            if i == 0:
                memo[0] = 1
                return 1
            rs = 1
            for j in range(i,-1,-1):
                # 因为前面的排序方式,使得这里不需要比较[0]了
                if envelopes[j][1] < envelopes[i][1]:
                    rs = max(rs, dp(j) + 1)
            memo[i] = rs
            return rs
        
        rs_list = []
        for i in range(len(envelopes)):
            rs_list.append(dp(i))
        return max(rs_list)

    # 53. 最大子序和 (opens new window)

    此题不好用滑动窗口,因为左边界什么时候移动不好确定

    查看答案 
    class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        memo = {}
        def dp(i):
            # 定义dp(i): 表示nums[:i+1]中,取nums[i]为最后子数组元素的最大和
            if i == 0:
                return nums[i]
            if i < 0:
                return 0
            if i in memo:
                return memo[i]
            # 两种情况:
            # 第一种: 如果dp(i-1)<0,则dp(i)就是nums[i]
            # 否则,就是dp(i) = nums[i] + dp(i+1)
            tmp= dp(i-1)
            if tmp < 0:
                rs = nums[i]
            else:
                rs = tmp + nums[i]
            memo[i] = rs
            return rs
        
        rs_list = [ dp(i) for i in range(len(nums))]
        # print(rs_list)
        return max(rs_list)

    # 1143. 最长公共子序列 (opens new window)

    查看答案 
    class Solution:
    def longestCommonSubsequence(self, text1: str, text2: str) -> int:
        memo = {}
        def dp(i,j):
            # 定义函数dp(i,j): 表示text1[:i+1] 和 test2[:j+1] 的最长公共子序列
            if i < 0 or j < 0:
                return 0
            if (i,j) in memo:
                return memo[(i,j)]
            # 两种情况
            # 第一种:如果text1[i] == text2[j],则
            if text1[i] == text2[j]:
                rs = dp(i-1,j-1) + 1
            # 第二种,text1[i] != text2[j]
            else:
                # 有两种情况:
                rs1 = dp(i-1,j)
                rs2 = dp(i,j-1)
                rs = max(rs1,rs2)
            memo[(i,j)] = rs
            return rs
        
        return dp(len(text1)-1, len(text2) - 1)

    # 583. 两个字符串的删除操作 (opens new window)

    此题和上题一样,先找出公共子序列, 然后用原长度减去公共长度,就是要操作的次数

    查看答案 
    class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        memo = {}
        def dp(i,j):
            # 定义函数dp(i,j): 表示word1[:i+1] 和 word2[:j+1] 的最长公共子序列长度
            if i < 0 or j < 0:
                return 0
            if (i,j) in memo:
                return memo[(i,j)]
            # 两种情况
            if word1[i] == word2[j]:
                rs = dp(i-1,j-1) + 1
            else:
                # 有两种情况:
                rs1 = dp(i-1,j)
                rs2 = dp(i,j-1)
                rs = max(rs1,rs2)
            memo[(i,j)] = rs
            return rs

        commen_length = dp(len(word1)-1, len(word2) - 1)

        return len(word1)+len(word2) - 2 * commen_length

    # 712. 两个字符串的最小ASCII删除和 (opens new window)

    查看答案 
    class Solution:
    def minimumDeleteSum(self, s1: str, s2: str) -> int:
        memo = {}
        def dp(i,j):
            #定义函数dp(i,j):表示使s1[:i+1] 和 s2[:j+1] 相等所需删除字符的ASCII值的最小和
            if i < 0:
                # 当i小于0时,需要把剩余的j删除
                rs = 0 
                for k in range(j+1):
                    rs += ord(s2[k])
                return rs
            if j < 0:
                rs = 0
                for k in range(i+1):
                    rs += ord(s1[k])
                return rs
            if (i,j) in memo:
                return memo[(i,j)]
            #两种情况
            if s1[i] == s2[j]:
                rs = dp(i-1,j-1)
            else:
                # 删除i
                rs1 = dp(i-1,j) + ord(s1[i])
                # 删除j
                rs2 = dp(i,j-1) + ord(s2[j])
                rs = min(rs1,rs2)
            memo[(i,j)] = rs
            return rs
        
        return dp(len(s1)-1, len(s2) -1 )

    # 516. 最长回文子序列 (opens new window)

    解法一: 会超时

    查看答案 
    class Solution:
    def longestPalindromeSubseq(self, s: str) -> int:
        memo = {}
        def dp(i,j):
            # 定义函数dp(i,j) 表示分别以s[i] 和 s[j] 为中间向两边扩散,得到的最长回文子序列长度
            if i < 0 or j > len(s) -1:
                return 0
            if (i,j) in memo:
                return memo[(i,j)]
            if s[i] == s[j]:
                if i != j:
                    rs = 2 + dp(i-1,j+1)
                else:
                    rs = 1 + dp(i-1,j+1)
            else:
                rs1 = dp(i-1,j)
                rs2 = dp(i,j+1)
                rs = max(rs1,rs2)
            memo[(i,j)] = rs
            return rs
        
        rs_list = []
        for k in range(len(s)):
            rs1 = dp(k,k)
            rs2 = dp(k,k+1)
            rs = max(rs1,rs2)
            rs_list.append(rs)
        # print(memo)
        return max(rs_list)

    class Solution:
    def longestPalindromeSubseq(self, s: str) -> int:
        memo = {}
        def dp(i,j):
            # 定义函数dp(i,j): 表示s[i:j+1]中最长的回文子序列的长度
            if i > j:
                return 0
            if i == j:
                return 1
            if (i,j) in memo:
                return memo[(i,j)]
            if s[i] == s[j]:
                rs = 2 + dp(i+1,j-1)
            else:
                rs1 = dp(i+1,j)
                rs2 = dp(i,j-1)
                rs = max(rs1, rs2)
            memo[(i,j)] = rs
            return rs

        return dp(0,len(s) - 1)
