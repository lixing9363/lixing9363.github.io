    class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        left = right = 0
        window  = {}
        rs = 0
        while right < len(s):
            add_char = s[right]
            right += 1
            window[add_char] = window[add_char] + 1 if add_char in window else 1
            while window[add_char] > 1:
                del_char = s[left]
                left += 1
                window[del_char] = window[del_char] - 1 
            rs = right - left if right-left > rs else rs
        
        return rs

    滑动窗口的过程:

    1. 先初始化一个left = right = 0作为索引,把索引区间[left,right)称为一个窗口window
    2. 一直增加right右边界的值,直到窗口内的数据符合need要求,停止增加right
    3. 再一直增加left的值,直到窗口内的数据不符合要求
    4. 再一直重复2.3两步,直到right到达右边终点

    需要注意点:

    1. 当right向右扩张时, 应该更新哪些数据? window , need ...

    2. 什么时候停止right增加,开始增加left? window中存在need中所有符合的数据?

    3. 当left向右移动时,应该更新哪些数据?window , need ...

    4. 我们需要的结果应该在什么时候更新?right右移时还是left右移时?

      一般是在进入left右移循环之后,left右移之前,进行判定后是否更新结果

    class Solution:
    def minWindow(self, s: str, t: str) -> str:
        need = {}
        for i in t:
            need[i]  = need[i] + 1 if i in need else 1
        
        left = right = 0
        window = {}
        char_need_count = 0

        lenth = len(s) + 1
        start = 0

        while right < len(s):
            add_chr = s[right]
            right += 1
            if add_chr in need:
                window[add_chr] = window[add_chr] + 1 if add_chr in window else 1
                if window[add_chr] == need[add_chr]:
                    char_need_count += 1
            while char_need_count == len(need):
                if right-left < lenth:
                    start = left
                    lenth = right - left
                del_char = s[left]
                left += 1
                if del_char in need:
                    window[del_char] = window[del_char] - 1
                    if window[del_char] < need[del_char]:
                        char_need_count -= 1
        
        if lenth == len(s) + 1:
            return ""
        return s[start:start+lenth]

    # 239. 滑动窗口最大值 (opens new window)

    如果用滑动窗口,每次窗口移动时,暴力求解窗口中的最大值,可以得出结果,但容易超时

    笔记

    此题用滑动窗口思路很简单，关键点在于找最大值容易超时

    那么就要用到单调队列的思想

    什么是单调队列（比如单调递减）：

    1. 这个队列最大的数在最前面，依次递减
    2. 当加入一个数时，从后面添加，如果加入的比后面的数大，则直接删除后面的数
    3. 当获取最大值时，直接返回最前面的数
    4. 当删除一个数，如果这个数是最前面的最大值，则删除，否则不做操作

    这样就使得这个队列不论我们怎么向里面添加元素，删除元素，最前面的值永远是最大的

    查看答案 
    class Que:
    def __init__(self):
        self.q = []
    
    def add(self,n):
        while self.q and n > self.q[-1]:
            self.q.pop()
        self.q.append(n)
    
    def get_max(self):
        if self.q:
            return self.q[0]
        return -1
    
    def remove(self,n):
        if self.q[0] == n:
            del self.q[0]

class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        left = right = 0

        q = Que()
        rs = []

        while right < len(nums):
            add_num = nums[right]
            right += 1
            q.add(add_num)
            
            while right - left == k:
                rs.append(q.get_max())
                del_num = nums[left]
                left += 1
                q.remove(del_num)
        return rs

    # 438. 找到字符串中所有字母异位词 (opens new window)

    查看答案 
    class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        need = {}
        for i in p:
            need[i] = need[i] + 1 if i in need else 1
        
        window = {}
        left = right = 0
        valid_char_count = 0
        rs = []
        while right < len(s):
            add_char = s[right]
            right += 1
            if add_char in need:
                window[add_char] = window[add_char] + 1 if add_char in window else 1
                if window[add_char] == need[add_char]:
                    valid_char_count += 1
            
            while right - left == len(p):
                if valid_char_count == len(need):
                    rs.append(left)
                del_char = s[left]
                left += 1
                if del_char in need:
                    if window[del_char] == need[del_char]:
                        valid_char_count -= 1
                    window[del_char] -= 1
        
        return rs

    # 567. 字符串的排列 (opens new window)

    查看答案 
    class Solution:
    def checkInclusion(self, s1: str, s2: str) -> bool:
        need = {}
        for i in s1:
            need[i] = need[i] + 1 if i in need else 1
        
        window = {}
        left = right = 0
        valid_char_count = 0

        while right < len(s2):
            add_char = s2[right]
            right += 1
            if add_char in need:
                window[add_char] = window[add_char] + 1 if add_char in window else 1
                if window[add_char] == need[add_char]:
                    valid_char_count += 1
            
            while right - left == len(s1):
                if valid_char_count == len(need):
                    return True
                del_char = s2[left]
                left += 1
                if del_char in need:
                    if window[del_char] == need[del_char]:
                        valid_char_count -= 1
                    window[del_char] -= 1
        
        return False
