    # 198. 打家劫舍 (opens new window)

    查看答案 
    class Solution:
    def rob(self, nums: List[int]) -> int:
        memo = {}
        def dp(i):
            # 定义函数dp(i): 表示在nums[:i]房屋中, 偷取了nums[i-1]时,偷到的最大金额
            if i <= 0:
                return 0
            if i == 1:
                return nums[0]
            if i in memo:
                return memo[i]
            # if i == 2:
            #     return max(nums[0],nums[1])
            # 两种情况,偷i-1, 不偷i-1
            rs1 = dp(i-2) +  nums[i-1]
            rs2 = dp(i-1)
            res = max(rs1,rs2)
            memo[i] = res
            return res
        
        rs = dp(len(nums))
        return rs

    # 213. 打家劫舍 II (opens new window)

    解法一:

    查看答案 
    class Solution:
    def rob(self, nums: List[int]) -> int:
        memo = {}
        def dp(i,rob_first):
            # 定义函数dp(i,rob_first): 表示在nums[:i]房屋中,rob_first表示是否偷取第一家, 偷取了nums[i-1]时,偷到的最大金额
            if i <= 0:
                return 0
            if i == 1:
                if rob_first:
                    return nums[0]
                return 0
            if (i,rob_first) in memo:
                return memo[(i,rob_first)]
            # 偷nums[i-1]时
            if i == len(nums) and  rob_first:
                rs1 = 0
            else:
                rs1 = dp(i-2,rob_first) + nums[i-1]
            # 不偷nums[i-1]时
            rs2 = dp(i-1,rob_first)
            res = max(rs1,rs2)
            memo[(i,rob_first)] = res
            return res
        rs = max(dp(len(nums),False),dp(len(nums),True))
        return rs

    def rob(self, nums: List[int]) -> int:
        if len(nums) == 1:
            return nums[0]
        # 如果偷第1家,则
        nums1 = nums[:-1]
        # 如果不偷第一家
        nums2 = nums[1:]
        return max(self.rob1(nums1),self.rob1(nums2))

    # 337. 打家劫舍 III (opens new window)

    查看答案 
    # Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rob(self, root: TreeNode) -> int:
        memo = {}
        def dp(node):
            # 定义函数dp(node): 表示以node节点为根开始偷的情况下,能偷到的最大值
            if not node:
                return 0
            if node in memo:
                return memo[node]
            left_node = node.left
            right_node = node.right
            # 第一种情况,偷node
            rs1 = node.val
            if left_node:
                rs1 += dp(left_node.left) + dp(left_node.right)
            if right_node:
                rs1 += dp(right_node.left) + dp(right_node.right)
            
            # 第二种情况, 不偷node
            rs2 = dp(left_node) + dp(right_node)
            res = max(rs1,rs2)
            memo[node] = res
            return res
        
        rs = dp(root)
        return rs
