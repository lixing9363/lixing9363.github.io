    vue 适配Markdown格式

    # vue 适配Markdown格式

    # 1. 安装mavonEditor
    $ npm install mavon-editor --save 或者
$ yarn add mavon-editor
    1
    2
    # 2. 在main.js中引入
    import mavonEditor from 'mavon-editor'
import 'mavon-editor/dist/css/index.css'
Vue.use(mavonEditor)
    1
    2
    3
    # 3. 使用
    <template>
    <div class="home">
        <div>
            <mavon-editor
                    :subfield="false"     // true： 双栏(编辑预览同屏)， false： 单栏(编辑预览分屏)
                    :toolbarsFlag="false" // 工具栏是否显示
                    defaultOpen="preview" //edit： 默认展示编辑区域 ， preview： 默认展示预览区域 , 其他 = edit
                    :editable="false"  // 是否允许编辑
                    v-model="value"/>
        </div>
    </div>
</template>
<script>
    export default {
        data() {
            return {
                value: "# iconfont图标本地使用\n" +
                    "\n" +
                    "##### 1. 选择图标添加进项目\n" +
                    "\n"
            }
        },
    }
</script>
